package org.uast;

/**
 * Checks if the node matches some structure, and extracts the data and children.
 *
 * @since 1.0
 */
public class Matcher0 implements Matcher {
    /**
     * The instance.
     */
    public static final Matcher INSTANCE = new Matcher0();

    /**
     * Expected node type.
     */
    private static final String EXPECTED_TYPE = "simpleIdentifier";

    /**
     * Expected number of child nodes.
     */
    private static final int EXPECTED_COUNT = 2;

    /**
     * Constructor.
     */
    private Matcher0() {
    }

    @Override
    public boolean match(final Node node,
        final Map<Integer, Node> children,
        final Map<Integer, String> data) {
        final boolean result = node.belongsToGroup(Matcher0.EXPECTED_TYPE)
            && node.getChildCount() == Matcher0.EXPECTED_COUNT;
        if (result) {
            children.put(1, node.getChild(0));
            children.put(2, node.getChild(1));
        }
        return result;
    }
}
