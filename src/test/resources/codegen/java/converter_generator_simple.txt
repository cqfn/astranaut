package org.uast;

import java.util.List;
import java.util.Map;
import java.util.TreeMap;
import org.cqfn.astranaut.core.algorithms.conversion.Converter;
import org.cqfn.astranaut.core.base.Builder;
import org.cqfn.astranaut.core.base.DummyNode;
import org.cqfn.astranaut.core.base.Factory;
import org.cqfn.astranaut.core.base.Node;

/**
 * Checks that the node matches the conversion rule:
 *   Variable -> Variable
 * and performs the conversion if possible.
 * @since 1.0
 */
public final class Rule0 implements Converter {
    /**
     * The instance.
     */
    public static final Converter INSTANCE = new Rule0();

    /**
     * The 'Variable' string.
     */
    private static final String VARIABLE = "Variable";

    /**
     * Constructor.
     */
    private Rule0() {
    }

    @Override
    public Node convert(final Node node, final Factory factory) {
        Node result = DummyNode.INSTANCE;
        final Map<Integer, List<Node>> children = new TreeMap<>();
        final Map<Integer, String> data = new TreeMap<>();
        final boolean matched = Matcher0.INSTANCE.match(node, children, data);
        if (matched) {
            result = Rule0.createRootNode(factory);
        }
        return result;
    }

    /**
     * Builds a node with 'Variable' type.
     * @param factory The node factory
     * @return A node
     */
    private static Node createRootNode(final Factory factory) {
        Node result = DummyNode.INSTANCE;
        final Builder builder = factory.createBuilder(Rule0.VARIABLE);
        if (builder.isValid()) {
            result = builder.createNode();
        }
        return result;
    }
}
